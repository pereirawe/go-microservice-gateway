upstream default_app {
    server gateway:8080;
}

upstream ms_payments_webhook {
    server 10.5.0.3:8088;
}

server {
    listen 80;
    listen [::]:80;
    server_name microservices.setuptecnologia.com.br;

    location / {
            #rewrite ^/(.*)/ /$1 permanent;
        return 301 https://$host$request_uri;
    }

    location ~ /.well-known/acme-challenge {
        root /tmp/acme_challenge;
    }

    location /nginx_status {
        stub_status on;
        
        # allow 127.0.0.1;
        # deny all;
    }
}

server {
    listen 443 ssl;
    server_name microservices.setuptecnologia.com.br;

    ssl_certificate /etc/letsencrypt/live/microservices.setuptecnologia.com.br/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/microservices.setuptecnologia.com.br/privkey.pem;

    location / {

            #if ($request_uri ~* "^/([^/]+)/?$") {1  
          #rewrite ^/(.*)/$ /$1 permanent;
            #}

        if (!-e $request_filename) {
            rewrite ^([^.]*[^/])$ $1/ permanent;
        }

        proxy_redirect          off;
        proxy_set_header        Host            $host;
        proxy_set_header        X-Real-IP       $remote_addr;
        proxy_set_header        X-Forwarded-For $proxy_add_x_forwarded_for;

        proxy_pass http://default_app;
    }

    location ~* /webhook {
        proxy_pass http://ms_payments_webhook;
       
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
